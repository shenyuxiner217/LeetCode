```javascript
var myPow = function(x, n) {
    if (x === 0) return 0;
    if (n === 0) return 1;
    const minus = n > 0 ? false : true;
    n = Math.abs(n);
    
    let power = 1;
    const powerArr = [[x, 1]]
    while (power * 2 <= n){
        x *= x;
        power *= 2;
        powerArr.push([x, power])
    }
    
    let result = 1;
    for (let i = powerArr.length - 1; i >= 0; i--) {
        while (powerArr[i][1] <= n) {
            n -= powerArr[i][1];
            result *= powerArr[i][0]
        }
        if (n === 0) break;
    }
    return minus ? 1 / result : result;
};
```
