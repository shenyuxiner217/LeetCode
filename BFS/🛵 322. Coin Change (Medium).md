var coinChange = function(coins, amount) {
    if (amount === 0) return 0;
    let set = new Set();
    let result = 0;
    let queue = [0];
    // coins.sort((a, b) => b - a)
    while (queue.length) {
        const temp = [];
        result++;
        while (queue.length) {
            const coin = queue.shift();
            for (let i = 0; i < coins.length; i++) {
                const sum = coin + coins[i];
                if (sum === amount) return result;
                if (sum < amount && !set.has(sum)) {
                    set.add(sum);
                    temp.push(sum);
                }
            }
        }
        queue = temp;
    }
    return -1;
};
