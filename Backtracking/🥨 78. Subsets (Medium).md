## [🥨 78. Subsets (Medium)](https://leetcode.com/problems/subsets/)


```javascript
var subsets = function(nums) {
    let result = [];
    function recur(temp, index) {
        // 这题的难点在于结束条件：
        // 由于for循环限制了temp的取值，也就限制了长度，因此不需要限制结束条件
        result.push([...temp]);
        for (let i = index; i < nums.length; i++){
            temp.push(nums[i])
            recur(temp, i + 1);
            temp.pop();
        }
    }
    recur([], 0);
    return result;
};
```
