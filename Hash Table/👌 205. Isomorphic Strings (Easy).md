var isIsomorphic = function(s, t) {
    // simple mapping can use object to solve instead of Map
    const s2t = {}
    const t2s = {}
    for(let i = 0; i < s.length; i ++) {
        if ((s2t[s[i]] && s2t[s[i]] != t[i]) || t2s[t[i]] && t2s[t[i]] != s[i] ){
            return false
        }
        s2t[s[i]] = t[i]
        t2s[t[i]] = s[i]
    }
    return true
    
};
